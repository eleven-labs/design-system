import { Meta } from '@storybook/addon-docs';
import { Layout, Table } from '@/storybook/components';
import { Text } from '@/components';
import { tokenVariables } from '@/constants';

<Meta
    title="Foundations/Design Tokens/Typography/FontFamily"
    parameters={{
        viewMode: 'docs',
        previewTabs: {
            canvas: { hidden: true }
        },
        backgrounds: {
            disable: true,
            grid: {
                disable: true,
            }
        },
    }}
/>

<Layout
    title="Typography"
    description="Font Family token scales"
>
    <Table
        title="Font Family"
        columns={[
            {
                name: 'name',
                label: 'Name',
            },
            {
                name: 'preview',
                label: 'Preview',
            },
            {
                name: 'weight_name',
                label: 'Weight Name',
            },
            {
                name: 'weight_value',
                label: 'Weight Value',
            }
        ]}
        rows={Object.entries(tokenVariables.asset.font).map(([tokenName, token]) => {
            const weight = Object.entries(tokenVariables['font-weight']).find(([_, tokenWeight]) => tokenWeight.value === token.weight)[0];
            return {
                name: (
                    <Text fontWeight="medium">{tokenName}</Text>
                ),
                preview: (
                    <Text
                        fontWeight={weight}
                    >
                        {tokenName}
                    </Text>
                ),
                weight_name: (
                    <Text>
                        {weight}
                    </Text>
                ),
                weight_value: (
                    <Text>
                        {token.weight}
                    </Text>
                ),
            }
        })}
    />
</Layout>
